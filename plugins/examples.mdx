## 示例 1: 计算器工具

```python
from typing import Any, Dict
from .base_tool import BaseTool, ToolDefinition

class CalculatorTool(BaseTool):
    
    def initialize(self, vector_store=None):
        pass
    
    @property
    def definition(self) -> ToolDefinition:
        return ToolDefinition(
            name="calculator",
            display_name="计算器",
            description="执行基本的数学计算（加、减、乘、除）",
            parameters={
                "type": "object",
                "properties": {
                    "operation": {
                        "type": "string",
                        "description": "运算类型：add, subtract, multiply, divide",
                        "enum": ["add", "subtract", "multiply", "divide"]
                    },
                    "a": {
                        "type": "number",
                        "description": "第一个数字"
                    },
                    "b": {
                        "type": "number",
                        "description": "第二个数字"
                    }
                },
                "required": ["operation", "a", "b"]
            }
        )
    
    async def execute(self, **kwargs) -> str:
        try:
            self.validate_parameters(kwargs)
            
            operation = kwargs.get("operation")
            a = kwargs.get("a")
            b = kwargs.get("b")
            
            result = 0
            op_symbol = ""
            
            if operation == "add":
                result = a + b
                op_symbol = "+"
            elif operation == "subtract":
                result = a - b
                op_symbol = "-"
            elif operation == "multiply":
                result = a * b
                op_symbol = "*"
            elif operation == "divide":
                if b == 0:
                    return "错误：除数不能为零"
                result = a / b
                op_symbol = "/"
            else:
                return f"错误：未知的运算类型 {operation}"
            
            return f"计算结果：{a} {op_symbol} {b} = {result}"
            
        except Exception as e:
            return f"计算过程中发生错误：{str(e)}"
```

## 示例 2: 天气查询工具

```python
import httpx
from typing import Any, Dict
from .base_tool import BaseTool, ToolDefinition

class WeatherTool(BaseTool):
    
    def __init__(self):
        self.api_key = None
    
    def initialize(self, vector_store=None):
        self.api_key = "your_api_key_here"
    
    @property
    def definition(self) -> ToolDefinition:
        return ToolDefinition(
            name="get_weather",
            display_name="天气查询",
            description="查询指定城市的实时天气信息",
            parameters={
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称（如：北京、上海）"
                    },
                    "units": {
                        "type": "string",
                        "description": "温度单位（celsius 或 fahrenheit）",
                        "enum": ["celsius", "fahrenheit"],
                        "default": "celsius"
                    }
                },
                "required": ["city"]
            }
        )
    
    async def execute(self, **kwargs) -> str:
        try:
            self.validate_parameters(kwargs)
            
            city = kwargs.get("city")
            units = kwargs.get("units", "celsius")
            
            async with httpx.AsyncClient() as client:
                params = {
                    "city": city,
                    "units": units,
                    "key": self.api_key
                }
                response = await client.get(f"https://api.weather.com/v1/current", params=params)
                response.raise_for_status()
                data = response.json()
            
            temperature = data.get("temp", "未知")
            condition = data.get("condition", "未知")
            humidity = data.get("humidity", "未知")
            
            return f"{city} 当前天气：\n温度：{temperature}°{'C' if units == 'celsius' else 'F'}\n天气：{condition}\n湿度：{humidity}%"
            
        except httpx.HTTPStatusError as e:
            return f"天气查询失败：HTTP {e.response.status_code}"
        except Exception as e:
            return f"天气查询过程中发生错误：{str(e)}"
```

## 示例 3: 数据库查询工具

```python
from typing import Any, Dict
from .base_tool import BaseTool, ToolDefinition
from database import SessionLocal
from models_db import User

class UserSearchTool(BaseTool):
    
    def initialize(self, vector_store=None):
        pass
    
    @property
    def definition(self) -> ToolDefinition:
        return ToolDefinition(
            name="user_search",
            display_name="用户搜索",
            description="搜索系统中的用户",
            parameters={
                "type": "object",
                "properties": {
                    "username": {
                        "type": "string",
                        "description": "用户名（可选，部分匹配）"
                    }
                }
            }
        )
    
    async def execute(self, **kwargs) -> str:
        username = kwargs.get("username", "")
        
        db = SessionLocal()
        try:
            query = db.query(User)
            if username:
                query = query.filter(User.username.contains(username))
            
            users = query.limit(10).all()
            
            if not users:
                return "未找到匹配的用户"
            
            result = "搜索结果：\n"
            for user in users:
                result += f"- {user.username} (ID: {user.id})\n"
            
            return result
        except Exception as e:
            return f"搜索用户时出错: {str(e)}"
        finally:
            db.close()
```

## 示例 4: 带缓存的工具

```python
from typing import Any, Dict
from .base_tool import BaseTool, ToolDefinition
from .cache import SearchResultCache
import httpx

class CachedSearchTool(BaseTool):
    
    def __init__(self):
        self._cache = SearchResultCache(max_size=500, ttl=1800)
    
    def initialize(self, vector_store=None):
        pass
    
    @property
    def definition(self) -> ToolDefinition:
        return ToolDefinition(
            name="cached_search",
            display_name="缓存搜索",
            description="带缓存功能的搜索工具",
            parameters={
                "type": "object",
                "properties": {
                    "query": {
                        "type": "string",
                        "description": "搜索查询"
                    }
                },
                "required": ["query"]
            }
        )
    
    async def execute(self, **kwargs) -> str:
        try:
            self.validate_parameters(kwargs)
            
            query = kwargs.get("query")
            
            cached = self._cache.get(query)
            if cached:
                return f"（缓存）{cached}"
            
            async with httpx.AsyncClient() as client:
                response = await client.get(f"https://api.example.com/search?q={query}")
                response.raise_for_status()
                data = response.json()
            
            result = f"搜索结果：{data.get('results', [])}"
            
            self._cache.set(query, result)
            
            return result
            
        except Exception as e:
            return f"搜索过程中发生错误：{str(e)}"
```

## 示例 5: 多参数工具

```python
from typing import Any, Dict
from .base_tool import BaseTool, ToolDefinition

class MultiParamTool(BaseTool):
    
    def initialize(self, vector_store=None):
        pass
    
    @property
    def definition(self) -> ToolDefinition:
        return ToolDefinition(
            name="multi_param_tool",
            display_name="多参数工具",
            description="演示多种参数类型的工具",
            parameters={
                "type": "object",
                "properties": {
                    "text": {
                        "type": "string",
                        "description": "文本内容"
                    },
                    "count": {
                        "type": "integer",
                        "description": "数量",
                        "minimum": 1,
                        "maximum": 100
                    },
                    "ratio": {
                        "type": "number",
                        "description": "比例",
                        "minimum": 0.0,
                        "maximum": 1.0
                    },
                    "enabled": {
                        "type": "boolean",
                        "description": "是否启用"
                    },
                    "tags": {
                        "type": "array",
                        "description": "标签列表",
                        "items": {
                            "type": "string"
                        }
                    },
                    "options": {
                        "type": "object",
                        "description": "配置选项",
                        "properties": {
                            "option1": {"type": "string"},
                            "option2": {"type": "boolean"}
                        }
                    }
                },
                "required": ["text"]
            }
        )
    
    async def execute(self, **kwargs) -> str:
        try:
            self.validate_parameters(kwargs)
            
            text = kwargs.get("text")
            count = kwargs.get("count", 1)
            ratio = kwargs.get("ratio", 0.5)
            enabled = kwargs.get("enabled", False)
            tags = kwargs.get("tags", [])
            options = kwargs.get("options", {})
            
            result = f"文本：{text}\n"
            result += f"数量：{count}\n"
            result += f"比例：{ratio}\n"
            result += f"启用：{enabled}\n"
            result += f"标签：{', '.join(tags)}\n"
            result += f"选项：{options}\n"
            
            return result
            
        except Exception as e:
            return f"处理过程中发生错误：{str(e)}"
```

## 注册示例

在 `backend/tools/__init__.py` 中注册这些工具：

```python
from .calculator import CalculatorTool
from .weather import WeatherTool
from .user_search import UserSearchTool
from .cached_search import CachedSearchTool
from .multi_param import MultiParamTool

_calculator_tool = CalculatorTool()
_weather_tool = WeatherTool()
_user_search_tool = UserSearchTool()
_cached_search_tool = CachedSearchTool()
_multi_param_tool = MultiParamTool()

def register_tools_from_database():
    enabled_tool_ids = load_tools_from_database()
    
    tools_to_register = {
        _calculator_tool.definition.name: _calculator_tool,
        _weather_tool.definition.name: _weather_tool,
        _user_search_tool.definition.name: _user_search_tool,
        _cached_search_tool.definition.name: _cached_search_tool,
        _multi_param_tool.definition.name: _multi_param_tool
    }
    
    for tool_name, tool_instance in tools_to_register.items():
        if tool_name in enabled_tool_ids:
            if not get_global_registry().has_tool(tool_name):
                register_tool(tool_instance)
        else:
            if get_global_registry().has_tool(tool_name):
                unregister_tool(tool_name)

register_tools_from_database()

def initialize_all_tools():
    _calculator_tool.initialize()
    _weather_tool.initialize()
    _user_search_tool.initialize()
    _cached_search_tool.initialize()
    _multi_param_tool.initialize()
```
